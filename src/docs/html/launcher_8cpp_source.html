21 void Launcher::popInfo(const char *_info, const uint16_t _time) {
22 const uint64_t beginTime = this->time;
23 bool onRender = true;
24
25 while (onRender) {

28 static float wPop = static_cast<float>(HAL::getFontWidth(_info)) + 2 * ui::cfg.popMargin; //宽度

30
31 static float yPop = 0 - hPop - 8; //从屏幕上方滑入
32 static float yPopTrg = 0;
33 if (time - beginTime < _time) yPopTrg = (static_cast<float>(sys::cfg.screenHeight) - hPop) / 3; //目标位置 中间偏上
34 else yPopTrg = 0 - hPop - 8; //滑出
35
36 static float xPop = (static_cast<float>(sys::cfg.screenWeight) - wPop) / 2; //居中
37
38 HAL::canvasClear();
39 /*渲染一帧*/
40 app->onLoop();
41 /*渲染一帧*/
42
43 HAL::setDrawType(0);
44 HAL::drawRBox(xPop - 4, yPop - 4, wPop + 8, hPop + 8, ui::cfg.popRadius + 2);
45 HAL::setDrawType(1); //反色显示
46 HAL::drawRFrame(xPop - 1, yPop - 1, wPop + 2, hPop + 2, ui::cfg.popRadius); //绘制一个圆角矩形
47 HAL::drawEnglish(xPop + ui::cfg.popMargin,
48 yPop + ui::cfg.popMargin + static_cast<float>(HAL::getFontHeight()), _info); //绘制文字
