42std::vector<std::string> wrapText(const std::string &text, const uint8_t screenWidth) {
43 std::vector<std::string> lines;
44 std::string currentLine;
45 currentLine.reserve(screenWidth + 1);
46
47 for (size_t i = 0; i < text.size(); ++i) {
48 currentLine += text[i];
49 if (HAL::getFontWidth(currentLine.c_str()) >= screenWidth || text[i] == '\n') {
50 lines.emplace_back(currentLine);
51 currentLine.clear();
52 }
53 }
54
55 if (!currentLine.empty()) {
56 lines.push_back(currentLine); // 添加剩余的部分
57 }
58
59 return lines;
60}
