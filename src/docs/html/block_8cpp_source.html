13void applyRotation(const float rotationMatrix[3][3], float points[8][3]) {
14 float np[8][3]{}; // 临时数组保存旋转后的坐标
15 for (int i = 0; i < 8; i++) {
16 for (int j = 0; j < 3; j++) {
17 float sum = 0;
18 for (int k = 0; k < 3; k++) {
19 sum += rotationMatrix[j][k] * points[i][k];
20 }
21 np[i][j] = sum;
22 }
23 }
24 // 更新原始点的坐标
25 for (int i = 0; i < 8; i++) {
26 for (int j = 0; j < 3; j++) {
27 points[i][j] = np[i][j];
28 }
29 }
30}
