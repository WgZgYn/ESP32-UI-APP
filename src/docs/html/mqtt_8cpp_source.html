35 void WiFiEvent(arduino_event_id_t event) {
36 Serial.printf("[WiFi-event] event: %d\n", event);
37 switch (event) {
38 case SYSTEM_EVENT_STA_GOT_IP:
39 Serial.println("WiFi connected");
40 Serial.println("IP address: ");
41 Serial.println(WiFi.localIP());
42 connectToMqtt();
43 break;
44 case SYSTEM_EVENT_STA_DISCONNECTED:
45 Serial.println("WiFi lost connection");
46 xTimerStop(mqttReconnectTimer, 0); // ensure we don't reconnect to MQTT while reconnecting to Wi-Fi
47#ifdef AUTO_CONNECT
48 xTimerStart(wifiReconnectTimer, 0);
49#endif
50 break;
51 default:
52 break;
53 }
54 }
