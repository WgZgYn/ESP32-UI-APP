104 void Camera::goToListItemRolling(const std::vector<float> &_posSelector) {
105 //这是一个让页面在一定情况下向下或向上滚动一行的函数
106 //当index向上超越了一个屏幕可以显示的内容 就要向上滚动一行 滚动到以当前选择项为第一项的页面
107 //当index向下超越了一个屏幕可以显示的内容 就要向下滚动一行 滚动到以当前选择项为最后一项的页面
108 //正常情况下 不移动
109
110 //如果不碰边界 不更新端点
111 //端点惰性赋值
112 //计算出每页第一行的index和最后一行的index
113 //碰到下边界就向下滚动 碰到上边界就向上滚动
114
115 //最开始左端点是0 右端点是max-1
116 //index超过右端点 就向下滚动index-右端点行 同时左右端点都加上index-右端点
117
118 static uint8_t direction = 0; //0: no roll, 1: up, 2: down
119
120 moving = true;
121 if (outOfView(_posSelector[0], _posSelector[1]) == 1) direction = 1;
122 if (outOfView(_posSelector[0], _posSelector[1]) == 2) direction = 2;
123
124 if (direction == 1) {
125 go(_posSelector[0], _posSelector[1]);
126 if (this->x == 0 - _posSelector[0] && this->y == 0 - _posSelector[1]) direction = 0;
127 }
128 if (direction == 2) {
129 go(_posSelector[0],
130 _posSelector[1] + ui::cfg.listLineHeight - static_cast<float>(sys::cfg.screenHeight));
131 if (this->x == 0 - _posSelector[0] && this->y == 0 - (
132 _posSelector[1] + ui::cfg.listLineHeight - static_cast<float>(sys::cfg.screenHeight)))
133 direction = 0;
134 }
135
136 if (!outOfView(_posSelector[0], _posSelector[1])) moving = false;
137 }
