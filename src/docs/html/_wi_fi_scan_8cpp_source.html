42void WiFiScanItem::onLoop() {
43 HAL::showInfo();
44
45 if (finish) {
46 HAL::delay(50);
47 return;
48 }
49
50 const int state = WiFi.scanComplete();
51 switch (state) {
52 case WIFI_SCAN_RUNNING:
53 static uint8_t time = 0;
54 if (time % 4 == 0) {
55 HAL::printInfo("|");
56 } else if (time % 4 == 1) {
57 HAL::printInfo("/");
58 } else if (time % 4 == 2) {
59 HAL::printInfo("-");
60 } else if (time % 4 == 3) {
61 HAL::printInfo("\\");
62 }
63 time = (time + 1) % 4;
64 HAL::delay(50);
65 HAL::cancelInfo();
66
67 break;
68 case WIFI_SCAN_FAILED:
69 inited = false;
70 break;
71 default:
72 HAL::printInfo("Scan Over");
73 nums = state;
74 inited = false;
75 finish = true;
76 break;
